1. LAMP 的安装

3. MYSQL 的安装

卸载 mysql

# 查看当前是否安装了mysql, 如果安装了, 先卸载
[root@xiamu ~]# rpm -qa | grep mysql
[root@xiamu ~]#

rpm -e mysql　　// 普通删除模式
rpm -e --nodeps mysql　　// 强力删除模式，如果使用上面命令删除时，提示有依赖的其它文件，则用该命令可以对其进行强力删除

安装 mysql
在 mysql 官网挑自己需要的版本https://downloads.mysql.com/archives/community/
这里使用的是 mysql8.30 版本

# 将近800多M, 喝杯茶抽根烟耐心等等
[root@xiamu opt]# wget https://downloads.mysql.com/archives/get/p/23/file/mysql-8.0.30-1.el7.x86_64.rpm-bundle.tar

# 解压
[root@xiamu opt]# tar -xvf mysql-8.0.30-1.el7.x86_64.rpm-bundle.tar
mysql-community-client-8.0.30-1.el7.x86_64.rpm
mysql-community-client-plugins-8.0.30-1.el7.x86_64.rpm
mysql-community-common-8.0.30-1.el7.x86_64.rpm
mysql-community-debuginfo-8.0.30-1.el7.x86_64.rpm
mysql-community-devel-8.0.30-1.el7.x86_64.rpm
mysql-community-embedded-compat-8.0.30-1.el7.x86_64.rpm
mysql-community-icu-data-files-8.0.30-1.el7.x86_64.rpm
mysql-community-libs-8.0.30-1.el7.x86_64.rpm
mysql-community-libs-compat-8.0.30-1.el7.x86_64.rpm
mysql-community-server-8.0.30-1.el7.x86_64.rpm
mysql-community-server-debug-8.0.30-1.el7.x86_64.rpm
mysql-community-test-8.0.30-1.el7.x86_64.rpm

# 实际上我们只需要使用到以下六个包
mysql-community-client-8.0.30-1.el7.x86_64.rpm
mysql-community-client-plugins-8.0.30-1.el7.x86_64.rpm
mysql-community-common-8.0.30-1.el7.x86_64.rpm
mysql-community-libs-8.0.30-1.el7.x86_64.rpm
mysql-community-server-8.0.30-1.el7.x86_64.rpm
mysql-community-icu-data-files-8.0.30-1.el7.x86_64.rpm

# 注意安装顺序, 不能打乱, 打乱就会出问题
# 顺序: common => client-plugins => libs => client => icu-data-files => server
[root@xiamu opt]# rpm -ivh mysql-community-common-8.0.30-1.el7.x86_64.rpm
[root@xiamu opt]# rpm -ivh mysql-community-client-plugins-8.0.30-1.el7.x86_64.rpm
[root@xiamu opt]# rpm -ivh mysql-community-libs-8.0.30-1.el7.x86_64.rpm

# 如果报错了, 请先移除mysql-libs, 然后再安装libs, 没有报错可以忽略这一步
[root@xiamu opt]# yum remove mysql-libs
[root@xiamu opt]# rpm -ivh mysql-community-libs-8.0.30-1.el7.x86_64.rpm

[root@xiamu opt]# rpm -ivh mysql-community-client-8.0.30-1.el7.x86_64.rpm
[root@xiamu opt]# rpm -ivh mysql-community-icu-data-files-8.0.30-1.el7.x86_64.rpm
[root@xiamu opt]# rpm -ivh mysql-community-server-8.0.30-1.el7.x86_64.rpm

# 安装完之后查看版本, 出现有内容, 说明安装成功
[root@xiamu opt]# mysql --version
mysql  Ver 8.0.30 for Linux on x86_64 (MySQL Community Server - GPL)

#mysql初始化
[root@xiamu opt]# mysqld --initialize --user=mysql

# 查看mysql密码, 细心找找[Note] [MY-010454] [Server] , 会有mysql默认的密码, 先记下来
[root@xiamu opt]# cat /var/log/mysqld.log
2022-12-24T11:10:00.649105Z 6 [Note] [MY-010454] [Server]
A temporary password is generated for root@localhost: 0sgZr(pv&d)q

#开启mysql服务
[root@xiamu opt]# systemctl status mysqld.service
[root@xiamu opt]# systemctl start mysqld.service
[root@xiamu opt]# systemctl status mysqld.service

# 设置mysql开机自启动
[root@xiamu opt]# systemctl list-unit-files | grep mysqld.service
mysqld.service                                enabled
# enabled表示开了自启动, disable表示关闭了自启动
# 如果显示的是disable, 可以使用下面的命令设置为enable
[root@xiamu opt]# systemctl disable mysqld.service
[root@xiamu opt]# systemctl enable mysqld.service


# 登录mysql
[root@xiamu opt]# mysql -uroot -p
Enter password:

#登录完之后需要修改密码
mysql> alter user 'root'@'localhost' identified by '123456';
Query OK, 0 rows affected (0.01 sec)

# 查看数据库
mysql> show databases;
+--------------------+
| Database           |
+--------------------+
| information_schema |
| mysql              |
| performance_schema |
| sys                |
+--------------------+
4 rows in set (0.00 sec)

mysql> use mysql;

# mysql默认是不允许root用户远程登录的
# 开启mysql远程登录
mysql> update user set host = '%' where user = 'root';
# %是通配符, 设置可以远程连接的ip, 单独设置一个%表示可以任意的ip远程连接root账户
# 更新完表之后需要刷新权限
mysql> flush privileges;
Query OK, 0 rows affected (0.00 sec)

# 再次修改密码, 这次修改之后, 可以使用在本地的电脑远程登录mysql
mysql> ALTER USER 'root'@'%' IDENTIFIED WITH mysql_native_password by '123456';
Query OK, 0 rows affected (0.01 sec)

# 退出mysql
mysql> exit

# 设置字符集是utf8
[root@xiamu ~]# vi /etc/my.cnf
# 在文件最后添加这三行
character_set_server=utf8
[mysqld]
default_authentication_plugin=mysql_native_password
[root@xiamu ~]# systemctl restart mysqld.service

# 关闭Windows和linux两边的防火墙
[root@xiamu ~]# systemctl status firewalld.service
[root@xiamu ~]# systemctl stop firewalld.service
[root@xiamu ~]# systemctl status firewalld.service
# 设置开机自启动防火墙关闭
[root@xiamu ~]# systemctl enable firewalld
[root@xiamu ~]# systemctl disable firewalld

至此 mysql 的安装就完成了
同样可以在自己的电脑上连接 mysql, 有两种连接方式, 一种是使用 SSH 连接, 另一种是 mysql 连接
填写 linux 账户密码, 点击测试连接, 显示连接成功

填写 mysql 账户密码, 点击测试连接, 显示连接成功

在 Mysql 中添加用户和设置用户权限


# 给root赋予授予用户权限的权限
mysql> GRANT system_user ON *.* TO 'root';

# 'root'@'host', @前面的是账户, @后面的是主机
# %是通配符, 设置可以远程连接的ip, 单独设置一个%表示可以任意的ip远程连接root账户
mysql> create user 'zhang3'@'%' identified by '123456';
Query OK, 0 rows affected (0.01 sec)
mysql> grant all privileges on *.* to 'zhang3'@'%';
Query OK, 0 rows affected (0.00 sec)

# 只让li4访问test数据库
mysql> create user 'li4'@'localhost' identified by '123456';
Query OK, 0 rows affected (0.00 sec)
mysql> grant all privileges on test.* to 'li4'@'localhost';

mysql> ALTER USER 'li4'@'localhost' IDENTIFIED WITH mysql_native_password by '123456';
mysql> ALTER USER 'zhang3'@'%' IDENTIFIED WITH mysql_native_password by '123456';

# 刷新权限
mysql> flush privileges;
Query OK, 0 rows affected (0.00 sec)

# 创建test数据库
mysql> create database test;
Query OK, 1 row affected (0.00 sec)

# 使用test数据库
mysql> use test;

4. 安装 PHP

[root@xiamu ~]# yum -y install php

5. LAMP 服务的整合

PHP 整合 Apache

[root@xiamu ~]# cd /var/www/html
[root@xiamu html]# touch v.php
[root@xiamu html]# vim v.php

# 编写如下代码
<?php
    echo '<title>hello world</title>';
    phpinfo();
?>

# Apache的状态/开启/关闭/重启
[root@xiamu html]# systemctl status httpd
[root@xiamu html]# systemctl start httpd
[root@xiamu html]# systemctl stop httpd
[root@xiamu html]# systemctl restart httpd

# 防火墙firewalld的状态/开启/关闭/重启
[root@xiamu html]# systemctl status firewalld
[root@xiamu html]# systemctl start firewalld
[root@xiamu html]# systemctl stop firewalld
[root@xiamu html]# systemctl restart firewalld

记得要重启 Apache 的, 关闭防火墙 firewalld
浏览器输入http://192.168.1.100/v.php

PHP 整合 MySql 和 Apache


# 安装php连接mysql的拓展包
[root@xiamu html]# yum install php-mysqli

[root@xiamu html]# cd /var/www/html/
[root@xiamu html]# vim show.php

# 复制如下php代码
<?php
$servername = "localhost";
$username = "li4";
$password = "123456";
$dbname = "test";

// 创建连接
$conn = new mysqli($servername, $username, $password, $dbname);
// Check connection
if ($conn->connect_error) {
    die("连接失败: " . $conn->connect_error);
}

$sql = "SELECT id, first, last, address, position FROM employees";
$result = $conn->query($sql);

if ($result->num_rows > 0) {
    // 输出数据
    while($row = $result->fetch_assoc()) {
        echo $row["id"].$row["first"] . " " . $row["last"] . $row["address"] . $row["position"]. "<br>";
    }
} else {
    echo "0 结果";
}
$conn->close();
?>


# 重启Apache
[root@xiamu html]# systemctl restart httpd

至此，LAMP 服务的安装与简单配置基本完成了。
