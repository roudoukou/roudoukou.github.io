后端核心代码:


@WebServlet("/select/student")
public class SelectStudentServlet extends ThymeleafViewServlet {

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        request.setCharacterEncoding("UTF-8");
        response.setCharacterEncoding("UTF-8");
        response.setContentType("application/json; charset=utf-8");

        String keyword = request.getParameter("keyword");

        StudentDAO studentDAO = new StudentDAOImpl();

        try {
            List<Student> list = studentDAO.selectStudentByIdOrName(keyword);
            HttpSession session = request.getSession();
            session.setAttribute("keyword", keyword);
            session.setAttribute("students", list);
            processTemplate("studentInfo", request, response);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

dao 代码:

public class StudentDAOImpl extends BaseDAO<Student> implements StudentDAO {
    @Override
    public List<Student> selectStudentByIdOrName(String keyword) throws SQLException {
        return executeQuery("select * from student where id = ? or name = ?",
                keyword, keyword);
    }
}
